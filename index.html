// ============================================== // 🔋 Battery Widget pro Scriptable // ============================================== // Stáhni Scriptable z App Store, vlož tento kód, // pak přidej Scriptable widget na plochu iPadu // a vyber tento skript. // // Podporuje: Small (1 zařízení), Medium (až 3), // Large (až 6 zařízení) // ============================================== // ====== KONFIGURACE ====== const CSV_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vTDOJ3L1eDoT0aBM58YzyjRwoAEOnIjTuohkhF-du1esLTyC2hWtztKcbBCQqXrrRu6aOia-bPrAfTn/pub?output=csv" // Kolik záznamů použít pro výpočet trendu const TREND_N = 6 // Historické průměry z analýzy (min/1%) - fallback const RATES = { active: 6, // min na 1% při aktivním použití standby: 57, // min na 1% ve standby charge: 1.2, // min na 1% nabíjení } // Barvy podle stavu const COLORS = { bgWidget: new Color("#1a2535"), bgRow: new Color("#1e2d3e"), // Nabíjení - zelená barCharging: new Color("#1a6a2a"), barChargingEnd: new Color("#2a8a3a"), // Normální (>40%) - teal/zelená barNormal: new Color("#1a5a3a"), barNormalEnd: new Color("#2a7a4a"), // Střední (20-40%) - žlutá/okrová barMedium: new Color("#6a5a1a"), barMediumEnd: new Color("#8a7a2a"), // Nízká (<20%) - červená barLow: new Color("#5a1a1a"), barLowEnd: new Color("#8a2a2a"), barLowBorder: new Color("#cc3333"), // Chyba/nedostupné - tmavá barError: new Color("#3a2a2a"), textWhite: Color.white(), textLight: new Color("#ddeedd"), textMuted: new Color("#5a7a6a"), textTime: new Color("#4a6a5a"), textLow: new Color("#ff6666"), textError: new Color("#6a5a5a"), bolt: Color.white(), } // ====== NAČTENÍ DAT ====== async function fetchRecords() { try { let req = new Request(CSV_URL) req.timeoutInterval = 15 let csv = await req.loadString() let rows = csv.split("\n").filter(r => r.trim()) let records = [] for (let i = 1; i < rows.length; i++) { let cols = parseCSVLine(rows[i]) if (cols.length < 3) continue let ts = parseDate(cols[0]) let lv = parseInt(cols[1]) let dev = cols[2].replace(/\n/g, "").trim() let chg = cols.length >= 4 && cols[3].trim().toUpperCase() === "TRUE" if (ts && !isNaN(lv)) { records.push({ ts, lv, dev, chg }) } } records.sort((a, b) => b.ts - a.ts) return records } catch (e) { console.error("Fetch error: " + e) return null } } function parseCSVLine(line) { let cols = [], cur = "", inQ = false for (let ch of line) { if (ch === '"') inQ = !inQ else if (ch === ',' && !inQ) { cols.push(cur.trim()); cur = "" } else if (ch === '\r') {} else cur += ch } cols.push(cur.trim()) return cols } function parseDate(s) { let p = s.match(/(\d+)\/(\d+)\/(\d+)\s+(\d+):(\d+):(\d+)/) if (!p) return null return new Date(+p[3], +p[1]-1, +p[2], +p[4], +p[5], +p[6]) } // ====== ANALÝZA ZAŘÍZENÍ ====== function analyzeDevice(recs) { if (!recs || !recs.length) return null let latest = recs[0] let recent = recs.slice(0, TREND_N) let trend = null if (recent.length >= 2) { let first = recent[0], last = recent[recent.length - 1] let dtMin = (first.ts - last.ts) / 60000 let dLv = first.lv - last.lv if (dtMin > 0 && dLv !== 0) { let pph = (dLv / dtMin) * 60 trend = { pph, minPerPct: dtMin / Math.abs(dLv) } } } let isCharging = latest.chg || (trend && trend.pph > 1) let isDischarging = !isCharging && trend && trend.pph < -0.3 // Interpolovaný odhad aktuálního % let now = new Date() let ageSec = (now - latest.ts) / 1000 let ageMin = ageSec / 60 let interpolated = latest.lv if (trend && ageMin > 2) { let estChange = (trend.pph / 60) * ageMin interpolated = Math.round(latest.lv + estChange) interpolated = Math.max(0, Math.min(100, interpolated)) } // Odhad zbývajícího času let estMin = null if (isCharging && trend && trend.pph > 0) { estMin = (100 - interpolated) * (60 / trend.pph) } else if (isDischarging && trend) { estMin = interpolated * (60 / Math.abs(trend.pph)) } else if (!isCharging && interpolated > 0) { // Fallback: standby rate estMin = interpolated * RATES.standby } // Stáří dat let stale = ageMin > 60 // starší než hodinu = potenciálně nedostupné let error = ageMin > 24 * 60 // starší než den = error return { name: prettifyName(latest.dev), rawName: latest.dev, level: interpolated, rawLevel: latest.lv, isCharging, isDischarging, estMin, trend, stale, error, lastUpdate: latest.ts, } } function prettifyName(dev) { let map = { "ipadpro": "iPad", "ipad": "iPad", "nord2": "Nord2", "g16": "G16", } return map[dev.toLowerCase().replace(/\s/g, "")] || dev } function formatEst(min) { if (min == null || min < 0) return null if (min < 1) return "<1m" let h = Math.floor(min / 60) let m = Math.round(min % 60) if (h >= 24) { let d = Math.floor(h / 24) return `~${d}d` } if (h > 0) return `~${h}h` return `~${m}m` } // ====== VYKRESLENÍ ====== // Barva pozadí řádku podle stavu function getBarColor(dev) { if (dev.error) return { start: COLORS.barError, end: COLORS.barError, border: null } if (dev.isCharging) return { start: COLORS.barCharging, end: COLORS.barChargingEnd, border: null } if (dev.level <= 15) return { start: COLORS.barLow, end: COLORS.barLowEnd, border: COLORS.barLowBorder } if (dev.level <= 40) return { start: COLORS.barMedium, end: COLORS.barMediumEnd, border: null } return { start: COLORS.barNormal, end: COLORS.barNormalEnd, border: null } } // Kreslení progress baru jako obrázek function drawBar(width, height, pct, colors, hasBorder) { let dc = new DrawContext() dc.size = new Size(width, height) dc.opaque = false dc.respectScreenScale = true let r = new Rect(0, 0, width, height) // Pozadí dc.setFillColor(COLORS.bgRow) let path = new Path() path.addRoundedRect(r, 14, 14) dc.addPath(path) dc.fillPath() // Výplň if (pct > 0) { let fillW = Math.max(width * pct / 100, 28) let fillR = new Rect(0, 0, fillW, height) dc.setFillColor(colors.start) let fillPath = new Path() fillPath.addRoundedRect(fillR, 14, 14) dc.addPath(fillPath) dc.fillPath() } // Okraj (pro nízkou baterii) if (hasBorder && colors.border) { dc.setStrokeColor(colors.border) dc.setLineWidth(2) let borderPath = new Path() borderPath.addRoundedRect(new Rect(1, 1, width - 2, height - 2), 13, 13) dc.addPath(borderPath) dc.strokePath() } return dc.getImage() } // === SMALL WIDGET (1 zařízení, velký displej) === function buildSmall(w, devices) { let dev = devices[0] if (!dev) { addError(w, "Žádná data"); return } let colors = getBarColor(dev) let bg = drawBar(300, 300, dev.level, colors, dev.level <= 15) w.backgroundImage = bg w.setPadding(14, 16, 14, 16) // Název nahoře let name = w.addText(dev.name) name.font = Font.boldSystemFont(16) name.textColor = new Color("#ffffffaa") w.addSpacer() // Velké procento uprostřed let row = w.addStack() row.centerAlignContent() if (dev.isCharging) { let bolt = row.addText("⚡") bolt.font = Font.boldSystemFont(36) } if (dev.error) { let q = row.addText("?") q.font = Font.boldMonospacedSystemFont(56) q.textColor = COLORS.textError } else { let pct = row.addText("" + dev.level) pct.font = Font.boldMonospacedSystemFont(56) pct.textColor = dev.level <= 15 ? COLORS.textLow : COLORS.textWhite } w.addSpacer() // Odhad času dole let est = formatEst(dev.estMin) if (est && !dev.error) { let info = w.addText(dev.isCharging ? `${est} do 100%` : `zbývá ${est}`) info.font = Font.mediumSystemFont(11) info.textColor = new Color("#ffffff66") } } // === MEDIUM/LARGE WIDGET (řádky) === function buildRows(w, devices, maxRows) { w.setPadding(10, 12, 10, 12) for (let i = 0; i < Math.min(devices.length, maxRows); i++) { if (i > 0) w.addSpacer(6) addDeviceRow(w, devices[i]) } } function addDeviceRow(w, dev) { let colors = getBarColor(dev) let rowH = 44 let rowW = 300 // Stack pro řádek let row = w.addStack() row.layoutHorizontally() row.centerAlignContent() row.size = new Size(0, rowH) row.setPadding(0, 0, 0, 0) // Bar jako pozadí row.backgroundImage = drawBar(rowW, rowH, dev.level, colors, dev.level <= 15) row.setPadding(0, 14, 0, 14) // Název let name = row.addText(dev.name) name.font = Font.boldSystemFont(15) name.textColor = dev.level <= 15 && !dev.isCharging ? COLORS.textLow : COLORS.textLight name.lineLimit = 1 row.addSpacer() // Odhad času let est = formatEst(dev.estMin) if (est && !dev.error) { let time = row.addText(est) time.font = Font.regularSystemFont(11) time.textColor = COLORS.textTime time.lineLimit = 1 row.addSpacer(8) } // Nabíjení bolt if (dev.isCharging) { let bolt = row.addText("⚡") bolt.font = Font.mediumSystemFont(13) row.addSpacer(4) } // Procenta / error if (dev.error) { let q = row.addText("?") q.font = Font.boldMonospacedSystemFont(22) q.textColor = COLORS.textError } else { let pct = row.addText("" + dev.level) pct.font = Font.boldMonospacedSystemFont(22) pct.textColor = dev.level <= 15 && !dev.isCharging ? COLORS.textLow : COLORS.textWhite } } function addError(w, msg) { let t = w.addText(msg) t.font = Font.regularSystemFont(14) t.textColor = COLORS.textError t.centerAlignText() } // ====== HLAVNÍ ====== async function main() { let w = new ListWidget() w.backgroundColor = COLORS.bgWidget let allRecords = await fetchRecords() if (!allRecords) { addError(w, "Chyba načítání") finalize(w) return } // Seskupit podle zařízení let byDev = {} allRecords.forEach(r => { let key = r.dev.toLowerCase().replace(/\s/g, "") if (!byDev[key]) byDev[key] = [] byDev[key].push(r) }) // Analyzovat každé zařízení let devices = [] for (let [key, recs] of Object.entries(byDev)) { let a = analyzeDevice(recs) if (a) devices.push(a) } // Seřadit: nabíjecí nahoru, pak podle % devices.sort((a, b) => { if (a.isCharging !== b.isCharging) return a.isCharging ? -1 : 1 return b.level - a.level }) // Vybrat layout podle velikosti widgetu let family = config.widgetFamily || "medium" if (family === "small") { buildSmall(w, devices) } else if (family === "medium") { buildRows(w, devices, 3) } else { buildRows(w, devices, 6) } finalize(w) } function finalize(w) { // Refresh za 15 min w.refreshAfterDate = new Date(Date.now() + 15 * 60 * 1000) Script.setWidget(w) Script.complete() // Preview při spuštění v appce if (!config.runsInWidget) { w.presentMedium() } } await main()
